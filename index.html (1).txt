let musicLoopTimeout=null;
function playMusicLoop(){
if(!musicPlaying||!audioCtx) return;
const vol=G.musicVol||0.4;
const t=audioCtx.currentTime;
const bpm=140;
const beat=60/bpm;
const bars=4;
const totalTime=bars*4*beat;

// Bass line
const bassNotes=[55,55,82,55,73,55,82,55];
bassNotes.forEach((freq,i)=>{
const o=audioCtx.createOscillator();const g=audioCtx.createGain();
o.connect(g);g.connect(audioCtx.destination);
o.type='sawtooth';o.frequency.value=freq;
const st=t+i*(beat*0.5);
g.gain.setValueAtTime(vol*0.07,st);g.gain.exponentialRampToValueAtTime(0.001,st+beat*0.45);
o.start(st);o.stop(st+beat*0.45);
});

// Hi-hat pattern
for(let i=0;i<16;i++){
const buf=audioCtx.createBuffer(1,audioCtx.sampleRate*0.05,audioCtx.sampleRate);
const d=buf.getChannelData(0);
for(let j=0;j<d.length;j++) d[j]=(Math.random()*2-1)*Math.exp(-j/(d.length*0.3));
const src=audioCtx.createBufferSource();src.buffer=buf;
const g=audioCtx.createGain();src.connect(g);g.connect(audioCtx.destination);
const st=t+i*(beat*0.25);
g.gain.setValueAtTime(vol*0.04,st);
src.start(st);
}

// Kick on beats 1 and 3
[0,2].forEach(b=>{
const o=audioCtx.createOscillator();const g=audioCtx.createGain();
o.connect(g);g.connect(audioCtx.destination);
o.type='sine';
const st=t+b*2*beat;
o.frequency.setValueAtTime(150,st);o.frequency.exponentialRampToValueAtTime(50,st+0.1);
g.gain.setValueAtTime(vol*0.25,st);g.gain.exponentialRampToValueAtTime(0.001,st+0.15);
o.start(st);o.stop(st+0.15);
});

// Snare on beats 2 and 4
[1,3].forEach(b=>{
const buf=audioCtx.createBuffer(1,audioCtx.sampleRate*0.1,audioCtx.sampleRate);
const d=buf.getChannelData(0);
for(let j=0;j<d.length;j++) d[j]=(Math.random()*2-1)*Math.exp(-j/(d.length*0.4));
const src=audioCtx.createBufferSource();src.buffer=buf;
const gn=audioCtx.createGain();src.connect(gn);gn.connect(audioCtx.destination);
const st=t+b*2*beat;
gn.gain.setValueAtTime(vol*0.12,st);
src.start(st);

// Snare tone
const o=audioCtx.createOscillator();const g2=audioCtx.createGain();
o.connect(g2);g2.connect(audioCtx.destination);
o.type='triangle';o.frequency.value=200;
g2.gain.setValueAtTime(vol*0.05,st);g2.gain.exponentialRampToValueAtTime(0.001,st+0.08);
o.start(st);o.stop(st+0.08);
});

// Melody line
const melody=[523,659,784,659,523,440,494,523];
melody.forEach((freq,i)=>{
if(i%2!==0) return;
const o=audioCtx.createOscillator();const g=audioCtx.createGain();
o.connect(g);g.connect(audioCtx.destination);
o.type='square';o.frequency.value=freq/2;
const st=t+i*beat;
g.gain.setValueAtTime(vol*0.025,st);g.gain.exponentialRampToValueAtTime(0.001,st+beat*0.7);
o.start(st);o.stop(st+beat*0.7);
});

musicLoopTimeout=setTimeout(()=>{if(musicPlaying) playMusicLoop();},totalTime*1000-50);
}

function fmt(n){
if(n>=1e15) return (n/1e15).toFixed(2)+'Qa';
if(n>=1e12) return (n/1e12).toFixed(2)+'T';
if(n>=1e9) return (n/1e9).toFixed(2)+'B';
if(n>=1e6) return (n/1e6).toFixed(2)+'M';
if(n>=1e3) return (n/1e3).toFixed(1)+'K';
return Math.floor(n).toString();
}

function getBuildingCost(b,owned){return Math.floor(b.baseCost*Math.pow(1.15,owned||0));}

const c=Math.floor(G.cookies);const cps=calcCPS();
document.getElementById('cookieDisplay').textContent=fmt(c);
document.getElementById('cpsDisplay').textContent=fmt(cps);
document.getElementById('allTimeDisplay').textContent=fmt(G.totalEarned);
document.getElementById('rebirthDisplay').textContent=G.rebirths;
document.getElementById('prestigeDisplay').textContent=G.prestigePoints+'✨';
document.getElementById('mainCookieNum').textContent=fmt(c);
document.getElementById('mainCPS').textContent=fmt(cps)+' / second';
document.getElementById('rebirthBonusDisplay').textContent=G.rebirthMultiplier.toFixed(1)+'x';
document.getElementById('prestigePointsDisplay').textContent=G.prestigePoints+' PP';
document.getElementById('settingNameDisplay').textContent=G.playerName;
document.getElementById('settingStats').textContent=`${fmt(G.totalClicks)} clicks | ${fmt(G.totalEarned)} earned | ${G.rebirths} rebirths`;
updateBuildingCards();updateUpgradeCards();updateNarutoCards();updatePrestigeCards();
}

function updateBuildingCards(){
BUILDINGS.forEach(b=>{
const card=document.getElementById('bcard-'+b.id);if(!card) return;
const owned=G.buildings[b.id]||0;const cost=getBuildingCost(b,owned);
const can=G.cookies>=cost;
card.querySelector('.building-count').textContent=owned;
card.querySelector('.building-cost').innerHTML=`<span style="color:${can?'var(--gold)':'var(--muted)'}">💰 ${fmt(cost)}</span>`;
const fill=card.querySelector('.progress-bar-fill');
if(fill) fill.style.width=Math.min(100,(G.cookies/cost)*100)+'%';
card.classList.toggle('can-afford',can);
});
}
function updateUpgradeCards(){
UPGRADES.forEach(u=>{
const card=document.getElementById('ucard-'+u.id);if(!card) return;
const purchased=G.upgrades.includes(u.id);
const can=!purchased&&G.cookies>=u.cost;
card.classList.toggle('can-afford',can);
const fill=card.querySelector('.upgrade-progress-fill');
if(fill&&!purchased) fill.style.width=Math.min(100,(G.cookies/u.cost)*100)+'%';
});
}
function updateNarutoCards(){
NARUTO_UPGRADES.forEach(u=>{
const card=document.getElementById('ncard-'+u.id);if(!card) return;
const purchased=G.narutoUpgrades.includes(u.id);
const can=!purchased&&G.cookies>=u.cost;
card.classList.toggle('can-afford',can);
const fill=card.querySelector('.upgrade-progress-fill');
if(fill&&!purchased) fill.style.width=Math.min(100,(G.cookies/u.cost)*100)+'%';
});
}
function updatePrestigeCards(){
PRESTIGE_UPGRADES.forEach(u=>{
const card=document.getElementById('pcard-'+u.id);if(!card) return;
const owned=G.prestigeUpgrades.includes(u.id);
const can=!owned&&G.prestigePoints>=u.cost;
card.classList.toggle('can-afford',can);
const btn=card.querySelector('.prestige-btn');
if(btn) btn.disabled=!can||owned;
});
}

function renderBuildings(){
const list=document.getElementById('buildingsList');list.innerHTML='';
BUILDINGS.forEach(b=>{
const owned=G.buildings[b.id]||0;const cost=getBuildingCost(b,owned);
const can=G.cookies>=cost;
const d=document.createElement('div');d.className='building-card'+(can?' can-afford':'');d.id='bcard-'+b.id;
d.innerHTML=`
<div class="building-header">
<div class="building-info">
<div class="building-name"><span class="building-icon">${b.icon}</span>${b.name}</div>
<div class="building-desc">${b.desc}</div>
<div><span class="building-cost"><span style="color:${can?'var(--gold)':'var(--muted)'}">💰 ${fmt(cost)}</span></span><span class="building-cps">+${b.baseCps}/s each</span></div>
</div>
<div class="building-count">${owned}</div>
</div>
<div class="progress-bar-wrap"><div class="progress-bar-fill" style="width:${Math.min(100,(G.cookies/cost)*100)}%"></div></div>
`;
d.addEventListener('click',()=>buyBuilding(b));
list.appendChild(d);
});
}

function buyBuilding(b){
const owned=G.buildings[b.id]||0;const cost=getBuildingCost(b,owned);
if(G.cookies<cost){showToast('Not Enough Cookies','Need '+fmt(cost),'');return;}
G.cookies-=cost;G.buildings[b.id]=(owned||0)+1;G.totalBuildings++;
playSFX('buy');updateUI();
showToast('Building Purchased!',b.icon+' '+b.name+' #'+(owned+1),'t-green');
checkAchievements();
}

function renderUpgrades(){
const list=document.getElementById('upgradesList');list.innerHTML='';
UPGRADES.forEach(u=>{
const purchased=G.upgrades.includes(u.id);
const can=!purchased&&G.cookies>=u.cost;
const d=document.createElement('div');d.className='upgrade-card'+(purchased?' purchased':can?' can-afford':'');d.id='ucard-'+u.id;
d.innerHTML=`
<div class="upgrade-name">${u.icon} ${u.name} ${purchased?'✓':''}</div>
<div class="upgrade-desc">${u.desc}</div>
<div class="upgrade-cost">${purchased?'✅ PURCHASED':'💰 '+fmt(u.cost)}</div>
<div class="upgrade-effect">${u.effect}</div>
${!purchased?`<div class="upgrade-progress-wrap"><div class="upgrade-progress-fill" style="width:${Math.min(100,(G.cookies/u.cost)*100)}%"></div></div>`:''}
`;
if(!purchased) d.addEventListener('click',()=>buyUpgrade(u));
list.appendChild(d);
});
}

function buyUpgrade(u){
if(G.upgrades.includes(u.id)) return;
if(G.cookies<u.cost){showToast('Not Enough Cookies','Need '+fmt(u.cost),'');return;}
G.cookies-=u.cost;G.upgrades.push(u.id);G.upgradesBought++;
playSFX('buy');recalcMults();renderUpgrades();updateUI();
showToast('Upgrade Unlocked! 🚀',u.icon+' '+u.name+': '+u.effect,'t-gold');
checkAchievements();
}

function renderNaruto(){
const list=document.getElementById('narutoList');list.innerHTML='';
NARUTO_UPGRADES.forEach((u,idx)=>{
const purchased=G.narutoUpgrades.includes(u.id);
const can=!purchased&&G.cookies>=u.cost;
// Lock upgrades after first if previous not purchased
const locked=idx>0&&!G.narutoUpgrades.includes(NARUTO_UPGRADES[idx-1].id)&&!purchased;
const d=document.createElement('div');
d.className='upgrade-card naruto-card'+(purchased?' purchased':locked?' locked-upgrade':can?' can-afford':'');
d.id='ncard-'+u.id;
d.setAttribute('draggable','false');
d.style.userSelect='none';
d.innerHTML=`
<div class="upgrade-name" style="color:${locked?'var(--muted)':'#ff6600'}">${locked?'🔒':u.icon} ${locked?'???':u.name}</div>
<div class="upgrade-desc">${locked?'Buy the previous upgrade to unlock this one.':u.desc}</div>
<div class="upgrade-cost" style="color:#ff4400">${purchased?'✅ ACTIVATED':locked?'🔒 LOCKED':'🔥 '+fmt(u.cost)}</div>
<div class="upgrade-effect" style="color:#ff6600">${locked?'':u.effect}</div>
${!purchased&&!locked?`<div class="upgrade-progress-wrap"><div class="upgrade-progress-fill" style="width:${Math.min(100,(G.cookies/u.cost)*100)}%;background:linear-gradient(90deg,#ff4400,#ff8800)"></div></div>`:''}
`;
if(!purchased&&!locked) d.addEventListener('click',()=>buyNarutoUpgrade(u));
list.appendChild(d);
});
}

function buyNarutoUpgrade(u){
if(G.narutoUpgrades.includes(u.id)) return;
if(G.cookies<u.cost){showToast('Not Enough Chakra 🦊','Need '+fmt(u.cost),'t-red');return;}
G.cookies-=u.cost;G.narutoUpgrades.push(u.id);G.upgradesBought++;
playSFX('buy');recalcMults();renderNaruto();updateUI();
showToast('🦊 Jutsu Activated!',u.icon+' '+u.name+': '+u.effect,'t-red');

}

function buyPrestige(uid){
const u=PRESTIGE_UPGRADES.find(x=>x.id===uid);if(!u) return;
if(G.prestigeUpgrades.includes(u.id)||G.prestigePoints<u.cost){showToast('Not Enough Prestige Points','Need '+u.cost+' PP','');return;}
G.prestigePoints-=u.cost;G.prestigeUpgrades.push(u.id);
playSFX('achievement');recalcMults();renderPrestige();updateUI();
showToast('✨ Prestige Upgrade!',u.icon+' '+u.name,'t-gold');
checkAchievements();
}

function renderRebirths(){
const list=document.getElementById('rebirthList');list.innerHTML='';
REBIRTHS.forEach(r=>{
const times=G.rebirthsDone.filter(x=>x===r.id).length;
const d=document.createElement('div');d.className='rebirth-card';
d.innerHTML=`
<div class="rebirth-header"><div class="rebirth-name">${r.name}</div><div class="rebirth-count">Done: ${times}x</div></div>
<div class="rebirth-desc">${r.desc}</div>
<div class="rebirth-effect">${r.effect}</div>
<div class="rebirth-cost">Cost: 💰 ${fmt(r.cost)} | Reward: +${r.prestigePts} Prestige Points</div>
<button class="rebirth-btn" onclick="doRebirth('${r.id}')" ${G.cookies<r.cost?'disabled':''}>REBIRTH →</button>
`;
list.appendChild(d);
});
}

function doRebirth(rid){
const r=REBIRTHS.find(x=>x.id===rid);
if(!r||G.cookies<r.cost) return;
if(!confirm(`Rebirth as "${r.name}"? Progress will reset!`)) return;
G.cookies-=r.cost;G.rebirthsDone.push(rid);G.rebirths++;
G.rebirthMultiplier*=r.mult;
G.prestigePoints+=r.prestigePts;
G.buildings={};G.totalBuildings=0;
if(!r.keepUp){G.upgrades=[];G.narutoUpgrades=[];G.upgradesBought=0;recalcMults();}
// Prestige start cookies
if(G.prestigeUpgrades.includes('p1')) G.cookies=100;
const fl=document.createElement('div');fl.className='rebirth-flash';document.body.appendChild(fl);setTimeout(()=>fl.remove(),1000);
playSFX('rebirth');
showToast('🔄 REBIRTH!',r.name+' — x'+G.rebirthMultiplier.toFixed(1)+' | +'+r.prestigePts+' PP','t-gold');
renderBuildings();renderUpgrades();renderNaruto();renderPrestige();renderRebirths();
recalcMults();updateUI();checkAchievements();
}

function renderAchievements(){
const list=document.getElementById('achievementsList');list.innerHTML='';
document.getElementById('achCount').textContent=G.achUnlocked.length+'/'+ACHIEVEMENTS.length+' unlocked';
ACHIEVEMENTS.forEach(a=>{
const u=G.achUnlocked.includes(a.id);
const d=document.createElement('div');d.className='achievement-card'+(u?' unlocked':'');
d.innerHTML=`<div class="achievement-icon">${u?a.icon:'🔒'}</div><div class="achievement-name">${u?a.name:'???'}</div><div class="achievement-desc">${u?a.desc:'Keep playing to unlock...'}</div><div class="achievement-req">${a.req}</div>`;
list.appendChild(d);
});
}

function checkAchievements(){
let nu=false;
ACHIEVEMENTS.forEach(a=>{
if(!G.achUnlocked.includes(a.id)&&a.check(G)){
G.achUnlocked.push(a.id);nu=true;
playSFX('achievement');
showToast('🏆 Achievement Unlocked!',a.icon+' '+a.name,'t-gold');
}
});
if(nu) renderAchievements();
}

function switchTab(name,btn){
document.querySelectorAll('.tab-content').forEach(t=>t.classList.remove('active'));
document.querySelectorAll('.tab-btn').forEach(b=>b.classList.remove('active'));
const tab=document.getElementById('tab-'+name);if(tab) tab.classList.add('active');
if(btn) btn.classList.add('active');
if(name==='rebirths') renderRebirths();
if(name==='prestige') renderPrestige();
}

function saveGame(){
G.lastSaveTime=Date.now();G.musicEnabled=musicPlaying;
localStorage.setItem('jynxziClicker_'+G.playerName,JSON.stringify(G));
showToast('💾 Saved!','Progress saved successfully.','t-green');
}

function resetProgress(){
if(!confirm('Are you SURE? This will permanently delete ALL your progress. This cannot be undone!')) return;
if(!confirm('Last chance! Permanently delete everything?')) return;
localStorage.removeItem('jynxziClicker_'+G.playerName);
location.reload();
}

function exportSave(){
const data=btoa(JSON.stringify(G));
navigator.clipboard.writeText(data).then(()=>showToast('📤 Exported!','Save data copied to clipboard.','t-green')).catch(()=>prompt('Copy this:',data));
}

function importSave(){
const raw=prompt('Paste your save data here:');
if(!raw) return;
try{
const sv=JSON.parse(atob(raw));Object.assign(G,sv);
recalcMults();renderBuildings();renderUpgrades();renderNaruto();renderPrestige();renderRebirths();renderAchievements();updateUI();
showToast('📥 Imported!','Save data loaded successfully.','t-green');
}catch(e){ showToast('❌ Import Failed','Invalid save data.','t-red'); }
}

function goToMenu(){
if(!confirm('Return to menu? Game will be saved.')) return;
saveGame();stopMusic();
clearInterval(gameTickInt);clearInterval(autoSaveInt);
document.removeEventListener('keydown',onKeyDown);document.removeEventListener('keyup',onKeyUp);

}
